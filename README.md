void setup()
{
  pinMode(7,INPUT);
  pinMode(6,OUTPUT);
  pinMode(4,OUTPUT);
}

void loop()
{
  digitalWrite(6,digitalRead(7));
  digitalWrite(4,digitalRead(7));
  
  delay(1000); // Wait for 1000 millisecond(s)
